250  /**
251  * @brief CRTP-class that provides additional pointer arithmetic operators synthesized from basic operators
252  * @details @copybrief
253  * Adds the following operators
254  * - postfix increment and decrement (requires the respective prefix operators)
255  * - array subscript operator[] (requires operator+ and dereference operator)
256  * - binary arithmetic operators (requires compound assignment operators)
257  * - inequality comparison (requires operator==)
258  * - less than or equal comparison (requires operator>)
259  * - greater than or equal comparison (requires operator<)
260  * @tparam Impl Base class
261  */
262  template<typename Impl>
263  struct SynthesizedOperators {
264 
265  /**
266  * Array subscript operator
267  * @tparam Implementation SFINAE helper, do not specify explicitly
268  * @param n index
269  * @return *(*this + n)
270  */
271  template<REQUIRES_IMPL(Impl, *(INSTANCE_OF_IMPL + INSTANCE_OF(typename Implementation::difference_type)))>
272  constexpr auto operator[](typename Implementation::difference_type n) const
273  noexcept(noexcept(*(std::declval<Impl>() + n))) {
274  return *(this->getImpl() + n);
275  }
276 
277  /**
278  * Postfix increment. Synthesized from prefix increment
279  * @tparam Implementation SFINAE helper, do not specify explicitly
280  * @return Instance of Impl
281  */
282  template<REQUIRES_IMPL(Impl, ++INSTANCE_OF_IMPL)>
283  constexpr Impl operator++(int)
284  noexcept(noexcept(++std::declval<Impl>()) && std::is_nothrow_copy_constructible_v<Impl>) {
285  auto tmp = this->getImpl();
286  this->getImpl().operator++();
287  return tmp;
288  }
289 
290  /**
291  * Postfix decrement. Synthesized from prefix decrement
292  * @tparam Implementation SFINAE helper, do not specify explicitly
293  * @return Instance of Impl
294  */
295  template<REQUIRES_IMPL(Impl, --INSTANCE_OF_IMPL)>
296  constexpr Impl operator--(int)
297  noexcept(noexcept(--std::declval<Impl>()) && std::is_nothrow_copy_constructible_v<Impl>) {
298  auto tmp = this->getImpl();
299  this->getImpl().operator--();
300  return tmp;
301  }
302 
303  /**
304  * Binary +plus operator. Synthesized from compound assignment operator+=
305  * @tparam Implementation SFINAE helper, do not specify explicitly
306  * @param it left hand side
307  * @param n right hand side
308  * @return Instance of Impl
309  */

394  /**
395  * @brief Class combining multiple iterators into one. Use it to iterate over multiple ranges at the same time.
396  * @details @copybrief
397  * ZipIterators only support the operators of the least powerful underling iterator. Zipping a random access
398  * iterator (e.g. from std::vector) and a bidirectional iterator (e.g. from std::list) results in a
399  * bidirectional iterator. All operators are SFINAE friendly.
400  *
401  * ZipIterators return a tuple of references to the range elements. When using
402  * structured bindings, no additional reference binding is necessary.
403  *
404  * Let ```z``` be a ZipIterator composed from two ```std::vector<int>```
405  * ```
406  * auto [val1, val2] = *z; // val1 and val2 are references to the vector elements
407  * val1 = 17; // this will change the respective value in the first vector
408  * ```
409  * @tparam Iterators Underlying iterator types
410  */
411  template<typename Iterators>
412  class ZipIterator
413  : public traits::iterator_category_from_value<traits::minimum_category_v<Iterators>>,
414  public SynthesizedOperators<ZipIterator<Iterators>> {
415 
416  public:
417  using value_type = traits::values_t<Iterators>;
418  using reference = value_type;
419  using pointer = void;
420  using difference_type = std::ptrdiff_t;
421 
422  private:
423  BINARY_TUPLE_FOR_EACH_FOLD(ELEMENT1 == ELEMENT2, ||, oneEqual)
424  BINARY_TUPLE_FOR_EACH_FOLD(ELEMENT1 < ELEMENT2, &&, allLess)
425  BINARY_TUPLE_FOR_EACH_FOLD(ELEMENT1 > ELEMENT2, &&, allGreater)
426  BINARY_TUPLE_FOR_EACH(std::min<difference_type>({ELEMENT1 - ELEMENT2 ...}), minDifference)
427  Iterators iterators;
428 
429  public:
430  using SynthesizedOperators<ZipIterator>::operator++;
431  using SynthesizedOperators<ZipIterator>::operator--;
432
